331:              * conversion happens (because we still want it to go through most

346:     /**
347:      * Moving a css file should update all relative urls.
348:      * Relative references (e.g. ../images/image.gif) in a certain css file,
349:      * will have to be updated when a file is being saved at another location
350:      * (e.g. ../../images/image.gif, if the new CSS file is 1 folder deeper).
351:      *
352:      * @param Converter $converter Relative path converter
353:      * @param string    $content   The CSS content to update relative urls for.
354:      *
355:      * @return string
356:      */
357:     protected function move(Converter $converter, $content)
358:     {
359:         /*
360:          * Relative path references will usually be enclosed by url(). @import
361:          * is an exception, where url() is not necessary around the path (but is
362:          * allowed).
363:          * This *could* be 1 regular expression, where both regular expressions
364:          * in this array are on different sides of a |. But we're using named
365:          * patterns in both regexes, the same name on both regexes. This is only
366:          * possible with a (?J) modifier, but that only works after a fairly
367:          * recent PCRE version. That's why I'm doing 2 separate regular
368:          * expressions & combining the matches after executing of both.
369:          */
370:         $relativeRegexes = array(
371:             // url(xxx)
372:             '/
373:             # open url()
374:             url\(
375: 
376:                 \s*
377: 
378:                 # open path enclosure
379:                 (?P<quotes>["\'])?
380: 
381:                     # fetch path
382:                     (?P<path>
383: 
384:                         # do not fetch data uris or external sources
385:                         (?!(
386:                             \s?
387:                             ["\']?
388:                             (data|https?):
389:                         ))
390: 
391:                         .+?
392:                     )
393: 
394:                 # close path enclosure
395:                 (?(quotes)(?P=quotes))
396: 
397:                 \s*
398: 
399:             # close url()
400:             \)
401: 
402:             /ix',
403: 
404:             // @import "xxx"
405:             '/
406:             # import statement
407:             @import
408: 
409:             # whitespace
410:             \s+
411: 
412:                 # we don\'t have to check for @import url(), because the
413:                 # condition above will already catch these
414: 
415:                 # open path enclosure
416:                 (?P<quotes>["\'])
417: 
418:                     # fetch path
419:                     (?P<path>
420: 
421:                         # do not fetch data uris or external sources
422:                         (?!(
423:                             ["\']?
424:                             (data|https?):
425:                         ))
426: 
427:                         .+?
428:                     )
429: 
430:                 # close path enclosure
431:                 (?P=quotes)
432: 
433:             /ix',
434:         );
435: 
436:         // find all relative urls in css
437:         $matches = array();
438:         foreach ($relativeRegexes as $relativeRegex) {
439:             if (preg_match_all($relativeRegex, $content, $regexMatches, PREG_SET_ORDER)) {
440:                 $matches = array_merge($matches, $regexMatches);
441:             }
442:         }
443: 
444:         $search = array();
445:         $replace = array();
446: 
447:         // loop all urls
448:         foreach ($matches as $match) {
449:             // determine if it's a url() or an @import match
450:             $type = (strpos($match[0], '@import') === 0 ? 'import' : 'url');
451: 
452:             // attempting to interpret GET-params makes no sense, so let's discard them for awhile
453:             $params = strrchr($match['path'], '?');

460:             $url .= $params;

462:             // build replacement
463:             $search[] = $match[0];

465:                 $replace[] = 'url('.$url.')';
466:             } elseif ($type == 'import') {
467:                 $replace[] = '@import "'.$url.'"';

469:         }
470: 
471:         // replace urls
472:         $content = str_replace($search, $replace, $content);
473: 
474:         return $content;

492:     /**
493:      * Shorthand 0 values to plain 0, instead of e.g. -0em.
494:      *
495:      * @param string $content The CSS content to shorten the zero values for.
496:      *
497:      * @return string
498:      */
499:     protected function shortenZeroes($content)
500:     {
501:         // reusable bits of code throughout these regexes:
502:         // before & after are used to make sure we don't match lose unintended
503:         // 0-like values (e.g. in #000, or in http://url/1.0)
504:         // units can be stripped from 0 values, or used to recognize non 0
505:         // values (where wa may be able to strip a .0 suffix)
506:         $before = '(?<=[:(, ])';
507:         $after = '(?=[ ,);}])';
508:         $units = '(em|ex|%|px|cm|mm|in|pt|pc|ch|rem|vh|vw|vmin|vmax|vm)';
509: 
510:         // strip units after zeroes (0px -> 0)
511:         // NOTE: it should be safe to remove all units for a 0 value, but in
512:         // practice, Webkit (especially Safari) seems to stumble over at least
513:         // 0%, potentially other units as well. Only stripping 'px' for now.
514:         // @see https://github.com/matthiasmullie/minify/issues/60
515:         $content = preg_replace('/'.$before.'(-?0*(\.0+)?)(?<=0)px'.$after.'/', '\\1', $content);
516: 
517:         // strip 0-digits (.0 -> 0)
518:         $content = preg_replace('/'.$before.'\.0+'.$units.'?'.$after.'/', '0\\1', $content);
519:         // strip trailing 0: 50.10 -> 50.1, 50.10px -> 50.1px
520:         $content = preg_replace('/'.$before.'(-?[0-9]+\.[0-9]+)0+'.$units.'?'.$after.'/', '\\1\\2', $content);
521:         // strip trailing 0: 50.00 -> 50, 50.00px -> 50px
522:         $content = preg_replace('/'.$before.'(-?[0-9]+)\.0+'.$units.'?'.$after.'/', '\\1\\2', $content);
523:         // strip leading 0: 0.1 -> .1, 01.1 -> 1.1
524:         $content = preg_replace('/'.$before.'(-?)0+([0-9]*\.[0-9]+)'.$units.'?'.$after.'/', '\\1\\2\\3', $content);
525: 
526:         // strip negative zeroes (-0 -> 0) & truncate zeroes (00 -> 0)
527:         $content = preg_replace('/'.$before.'-?0+'.$units.'?'.$after.'/', '0\\1', $content);
528: 
529:         return $content;
530:     }

553:      * Strip whitespace.
554:      *
555:      * @param string $content The CSS content to strip the whitespace for.
556:      *
557:      * @return string
558:      */
559:     protected function stripWhitespace($content)
560:     {
561:         // remove leading & trailing whitespace
562:         $content = preg_replace('/^\s*/m', '', $content);
563:         $content = preg_replace('/\s*$/m', '', $content);
564: 
565:         // replace newlines with a single space
566:         $content = preg_replace('/\s+/', ' ', $content);
567: 
568:         // remove whitespace around meta characters
569:         // inspired by stackoverflow.com/questions/15195750/minify-compress-css-with-regex
570:         $content = preg_replace('/\s*([\*$~^|]?+=|[{};,>~]|!important\b)\s*/', '$1', $content);
571:         $content = preg_replace('/([\[(:])\s+/', '$1', $content);
572:         $content = preg_replace('/\s+([\]\)])/', '$1', $content);
573:         $content = preg_replace('/\s+(:)(?![^\}]*\{)/', '$1', $content);
574: 
575:         // whitespace around + and - can only be stripped in selectors, like
576:         // :nth-child(3+2n), not in things like calc(3px + 2px) or shorthands
577:         // like 3px -2px
578:         $content = preg_replace('/\s*([+-])\s*(?=[^}]*{)/', '$1', $content);
579: 
580:         // remove semicolon/whitespace followed by closing bracket
581:         $content = str_replace(';}', '}', $content);
582: 
583:         return trim($content);
584:     }
